<script>

function decode(){

  let serial = document.getElementById("serial").value.toUpperCase();

  let result = document.getElementById("result");


  // -------------------

  // JERE FORMAT

  // -------------------

  if(/^\d{3}[WJ][A-L][GHI]$/.test(serial)){

    let build = serial.substring(0,3);

    let monthCode = serial.charAt(4);

    let yearCode = serial.charAt(5);


    let months = {

      A:"January",B:"February",C:"March",D:"April",

      E:"May",F:"June",G:"July",H:"August",

      I:"September",J:"October",K:"November",L:"December"

    };


    let years = {G:"1977",H:"1978",I:"1979"};


    result.innerHTML =

      "Manufacturer: Jere<br>" +

      "Year: "+years[yearCode]+"<br>" +

      "Month: "+months[monthCode]+"<br>" +

      "Production Number: "+build+"<br>" +

      "Serial Format: Jere Fabrication (1977–1979)";

    return;

  }


  // -------------------

  // KOIZUMI VM FORMAT

  // -------------------

  if(/^VM\d{1,4}$/.test(serial)){

    let build = serial.substring(2);


    result.innerHTML =

      "Manufacturer: Koizumi<br>" +

      "Year: 1979<br>" +

      "Production Period: March–June<br>" +

      "Production Number: "+build+"<br>" +

      "Serial Format: Koizumi VM Series";

    return;

  }


  // -------------------

  // KOIZUMI DATE FORMAT

  // -------------------

  if(/^[901234][A-Z]\d{4}[A-Z]?$/.test(serial)){

    let yearCode = serial.charAt(0);

    let monthCode = serial.charAt(1);

    let build = serial.substring(2,6);


    let years = {

      9:"1979",0:"1980",1:"1981",

      2:"1982",3:"1983",4:"1984"

    };


    let months = {

      A:"January",B:"February",C:"March",D:"April",

      E:"May",F:"June",G:"July",H:"August",

      X:"September",J:"October",K:"November",L:"December"

    };


    result.innerHTML =

      "Manufacturer: Koizumi<br>" +

      "Year: "+years[yearCode]+"<br>" +

      "Month: "+months[monthCode]+"<br>" +

      "Production Number: "+build+"<br>" +

      "Serial Format: Koizumi Date Code";

    return;

  }


  // -------------------

  // FAIRLY FORMAT (updated with C for 1980)

  // -------------------

  if(/^F[0-5C]\d{2}\d{4}$/.test(serial)){

    let yearCode = serial.charAt(1);

    let month = serial.substring(2,4);

    let build = serial.substring(4);


    // convert backwards C to 0

    if(yearCode === "C"){

      yearCode = "0";

    }


    let years = {

      0:"1980",1:"1981",2:"1982",

      3:"1983",4:"1984",5:"1985"

    };


    let monthNames = {

      "01":"January","02":"February","03":"March",

      "04":"April","05":"May","06":"June",

      "07":"July","08":"August","09":"September",

      "10":"October","11":"November","12":"December"

    };


    result.innerHTML =

      "Manufacturer: Fairly<br>" +

      "Year: "+years[yearCode]+"<br>" +

      "Month: "+monthNames[month]+"<br>" +

      "Production Number: "+build+"<br>" +

      "Serial Format: Fairly Date Code (1980–1985)";

    return;

  }


  // -------------------

  // Not recognized

  // -------------------

  result.innerHTML = "Serial number not recognised.";

}
